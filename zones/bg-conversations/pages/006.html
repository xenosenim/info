четене и писане

аз чета.

я читаю.

аз чета една буква.

я читаю букву.

аз чета една дума.

я читаю слово.

аз чета едно изречение.

я читаю предложение.

аз чета едно писмо.

я читаю письмо.

аз чета една книга.

я читаю книгу.

аз чета.

я читаю.

ти четеш.

ты читаешь.

той чете.

он читает.

аз пиша.

я пишу.

аз пиша една буква.

я пишу букву.

аз пиша една дума.

я пишу слово.

аз пиша едно изречение.

я пишу предложение.

аз пиша едно писмо.

я пишу письмо.

аз пиша една книга.

я пишу книгу.

аз пиша.

я пишу.

ти пишеш.

ты пишешь.

той пише.

он пишет.
